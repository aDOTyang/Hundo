The Code

                    
                        function getValues() {
  let startValue = document.getElementById("startValue").value;
  let endValue = document.getElementById("endValue").value;

  // string values from HTML need to be parsed into numbers
  startValue = parseInt(startValue);
  endValue = parseInt(endValue);

  if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
    let numbers = generateNumbers(startValue, endValue);
  
  displayNumbers(numbers);
  } else {
    Swal.fire({
      icon: "error",
      title: "NO HUNDO",
      text: "Only integers are allowed for HUNDO!",
    });
  }
}

function generateNumbers(startValue, endValue) {
  let numbers = [];

  for (let i = startValue; i <= endValue; i++) {
    numbers.push(i);
  }
  return numbers;
}

function displayNumbers(numbers) {
  let className = "even";
  let templateRows = "";
  for (let i = 0; i < numbers.length; i++) {
    let number = numbers[i];
    if (number % 2 == 0) {
      className = "even";
    } else {
      className = "odd";
    }

    if (i % 10 == 0) {
      templateRows += "";
    }

    templateRows += `${number}`;

    if ((i + 1) % 10 == 0) {
      templateRows += "";
    }
  }
  document.getElementById("results").innerHTML = templateRows;
}

The code is structured in three function per typical MVC convention.

getValues()

The getValues() function pulls in the inputs startValue and endValue from the HTML form fields and parses it into a usable integer. Once the conditions have been verified, the getValues() function then calls the generateNumbers() and displayNumbers() functions which will calculate the intended result and format the table, respectively.

generateNumbers()

The generateNumbers() function takes the input values from getValues() and simply pushes them into an array using single increments.

displayNumbers()

The displayNumbers() function checks the result for even/odd output and styles it accordingly using custom CSS. It then further styles the table so that it inserts a new table row where i is a multiple of 10 for easier viewing. If i is not a multiple of 10, displayNumbers() simply tells the HTML to add i to the end of the existing table row.
